# Docs for the Azure Web Apps Deploy action: https://github.com/Azure/webapps-deploy
# More GitHub Actions for Azure: https://github.com/Azure/actions

name: Build and deploy Node.js app to Azure Web App - hijri

on:
  workflow_dispatch:  # Manual trigger only; GitHub Pages is the primary deployment

jobs:
  build:
    runs-on: ubuntu-latest
    permissions:
      contents: read #This is required for actions/checkout

    steps:
      - uses: actions/checkout@v4

      - name: Set up Node.js version
        uses: actions/setup-node@v3
        with:
          node-version: '22.x'

      - name: npm install, build, and test
        run: |
          npm install
          npm run build --if-present
          npm run test --if-present

      - name: Verify build output exists
        run: |
          test -f apps/web/dist/index.html
          ls -la apps/web/dist

      - name: Upload artifact for deployment job
        uses: actions/upload-artifact@v4
        with:
          name: web-dist
          path: apps/web/dist

  deploy:
    runs-on: ubuntu-latest
    needs: build
    permissions:
      id-token: write #This is required for requesting the JWT
      contents: read #This is required for actions/checkout

    steps:
      - name: Download artifact from build job
        uses: actions/download-artifact@v4
        with:
          name: web-dist
          path: dist

      - name: Resolve deploy package path
        id: resolve-package
        run: |
          if [ -f dist/index.html ]; then
            echo "package_path=dist" >> "$GITHUB_OUTPUT"
          elif [ -f dist/apps/web/dist/index.html ]; then
            echo "package_path=dist/apps/web/dist" >> "$GITHUB_OUTPUT"
          else
            echo "Unable to find index.html in downloaded artifact"
            find dist -maxdepth 6 -type f | head -n 200
            exit 1
          fi

          echo "Using package path: $(cat $GITHUB_OUTPUT | grep package_path | cut -d'=' -f2-)"
      
      - name: Login to Azure
        uses: azure/login@v2
        with:
          client-id: ${{ secrets.AZUREAPPSERVICE_CLIENTID_7FB3033BB95C40C7B9E382C5BC8CE3A5 }}
          tenant-id: ${{ secrets.AZUREAPPSERVICE_TENANTID_79E5E3553B8B43FAB0E91B1159CC1AA1 }}
          subscription-id: ${{ secrets.AZUREAPPSERVICE_SUBSCRIPTIONID_FF49F7EB31B84DB6BB39D0149BFCFF73 }}

      - name: 'Deploy to Azure Web App'
        id: deploy-to-webapp
        uses: azure/webapps-deploy@v3
        with:
          app-name: 'hijri'
          slot-name: 'Production'
          package: ${{ steps.resolve-package.outputs.package_path }}

      - name: Configure startup command for static SPA
        run: |
          az webapp config set \
            --resource-group hijri-calculator \
            --name hijri \
            --startup-file "pm2 serve /home/site/wwwroot --no-daemon --spa"
